\question[15]
Consider the system of linear differential equations
\begin{align*}
\frac{dx}{dt}=&-3x+y\\
\frac{dy}{dt}=&100(2x-y)
\end{align*}
\begin{parts}
\part What is the quasi-equilibrium hypothesis for this system?
\part Use the quasi-equilibrium hypothesis to eliminate $y$ and show that the solution of the simplified system is
\begin{align*}
x(t)=& exp^{-t}\\
y(t)=& 2 exp^{-t}
\end{align*}
\end{parts}


\question[8]
Assume that an insect population, $x(t)$, is controlled by a natural predator population, $y(t)$.
\begin{parts}
\part Write a model describing the interaction of the 2 populations.
\part Assume that an insecticide is used to reduce the population of insects, but it is also toxic to the predators; hence, the poison kills both predatr and prey at rates proportional to their respective populations. Modify your model from \emph{(a)}.
\end{parts}
